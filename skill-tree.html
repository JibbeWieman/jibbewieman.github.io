using Unity.FPS.Game;
using UnityEngine;

public class XpManager : Manager
{
                                    #region VARIABLES

                                    public int PlayerLevel { get; private set; }
                                    public int PlayerXP { get; private set; }

    [Tooltip("Previous XP Threshold required to level up")]
                                    public int PrevThreshold { get; private set; }

    [Tooltip("XP required to level up")]
    public int CurThreshold { get; private set; }

    #endregion

    #region UNITY METHODS

    public override void Start()
    {
        // Initialise XP system and subscribe to XP update events
        base.Start();

        EventManager.AddListener<XpUpdateEvent>(UpdateXP);

        PlayerLevel = 1;
        PlayerXP = 0;
        PrevThreshold = 0;

        CalculateXPThreshold();
    }

        private void OnDestroy()
    {
        // Unsubscribe from XP update events
        EventManager.RemoveListener<XpUpdateEvent>(UpdateXP);
    }

    #endregion

    #region XP METHODS

    // Updates the player's XP based on the event data
    public void UpdateXP(XpUpdateEvent evt)
    {
        PlayerXP = evt.XP;

            if (PlayerXP >= CurThreshold)
        {
            LevelUpEvent lvlUpEvt = Events.LevelUpEvent;
            LevelUp(lvlUpEvt);
        }
    }

    // Handles logic for when the player levels up
    private void LevelUp(LevelUpEvent evt)
    {
        evt.Level++;

        PlayerLevel = evt.Level;

        SkillPointsChangedEvent skillPointEvent = Events.SkillPointsChangedEvent;
        skillPointEvent.SkillPoints++;

        CalculateXPThreshold();
    }

        // Calculates the next XP threshold for levelling up
        private void CalculateXPThreshold()
    {
        PrevThreshold = CurThreshold;
        CurThreshold = PlayerLevel * 150;
    }

                                    #endregion
}


using System.Collections;
using TMPro;
using Unity.FPS.Game;
using UnityEngine;
using UnityEngine.UI;

namespace Unity.FPS.UI
{
    public class PlayerXpBar : MonoBehaviour
    {
    #region VARIABLES

    [Tooltip("Image component displaying current XP")]
    public Image XpFillImage;

    public TextMeshProUGUI PlayerLevelText;

    [SerializeField, Tooltip("Speed of the XP bar transition")]
    private float LerpSpeed = 5f;

    [SerializeField]
    private SceneTypeObject ST_GameManager;

    private XpManager _xpManager;
    private Coroutine currentLerpCoroutine;

    #endregion

    #region UNITY METHODS

    // Initialises XP Manager and sets up XP bar on start
    void Start()
    {
    _xpManager = ST_GameManager.Objects[0].GetComponent<XpManager>();

    DebugUtility.HandleErrorIfNullGetComponent<XpManager, PlayerXpBar>(_xpManager, this, gameObject);

    EventManager.AddListener<XpUpdateEvent>(UpdateXpBar);

    XpFillImage.fillAmount = 0;
    }

    // Cleans up event subscriptions when destroyed
    private void OnDestroy()
    {
    EventManager.RemoveListener<XpUpdateEvent>(UpdateXpBar);
    }

    #endregion

    #region XP METHODS

    // Updates the XP bar fill amount and level text
    private void UpdateXpBar(XpUpdateEvent evt)
    {
    float targetXpFill = (float)(_xpManager.PlayerXP - _xpManager.PrevThreshold)
    / (float)(_xpManager.CurThreshold - _xpManager.PrevThreshold);

    if (currentLerpCoroutine != null)
    {
    StopCoroutine(currentLerpCoroutine);
    }

    currentLerpCoroutine = StartCoroutine(LerpXpFill(targetXpFill));
    PlayerLevelText.text = _xpManager.PlayerLevel.ToString();
    }

    // Lerps the fill amount of the XP bar towards the target value
    private IEnumerator LerpXpFill(float targetXpFill)
    {
    while (Mathf.Abs(XpFillImage.fillAmount - targetXpFill) > 0.01f)
    {
    XpFillImage.fillAmount = Mathf.Lerp(XpFillImage.fillAmount, targetXpFill, Time.deltaTime * LerpSpeed);
    yield return null;
    }

    XpFillImage.fillAmount = targetXpFill;
    }

    #endregion
    }
    }
